## üöÄ Usage

### Installation

```bash
# All dependencies included in monorepo
pnpm install
```

### Import Components

```tsx
import {
  DataFortress,
  Afanda,
  ComparisonCockpit,
  MagicPaste,
  ExceptionFirstMode,
  SmartQueueWorkbench,
  CommandKPalette,
  CRUDSPToolbar,
  AuditableField,
  PolicyAwareUI,
  // ... and 45 more
} from '@workspace/shared-ui/blocks';
```

### Quick Start

```tsx
// Customer Management Table
<DataFortress
  data={customers}
  columns={customerColumns}
  selectable
  auditEnabled
  exportable
/>

// Team Collaboration Hub
<Afanda
  roomId="sales-team"
  roomName="Sales Team"
/>

// Document Comparison
<ComparisonCockpit
  documents={[invoice, purchaseOrder]}
  mode="side-by-side"
/>

// Exception Management
<ExceptionFirstMode
  items={orderExceptions}
  onResolve={handleResolve}
/>
```
